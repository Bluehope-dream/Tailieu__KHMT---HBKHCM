void loop (){
   digitalWrite(12, LOW); 
  digitalWrite(13, HIGH); // đẩy điện ra mức cao ở chân 13
   delay(2000); // hàm độ trễ delay 1000~ 1s 
   digitalWrite(13,LOW); // Đẩy điện chân 13 về mức 0
  digitalWrite(12, HIGH); 
  delay(2000);
}
